# Single condition / Plane

#!/bin/bash -l
#SBATCH --job-name=mrnet_eval
# Request 1 node
#SBATCH -N 1
# GPU partition
#SBATCH --partition=gpu
# Request 1 GPU 
#SBATCH --gres=gpu:1
# Number of CPU cores
#SBATCH --ntasks-per-node=4
# Expected runtime: 30 minutes 
#SBATCH -t 00:30:00
# Email notifications
#SBATCH --mail-type=ALL
#SBATCH --mail-user=student@ucdconnect.ie
# Output logs
#SBATCH --output=eval_output/eval_%j.log
#SBATCH --error=eval_output/eval_%j.err

# Print job information
echo "=========================================="
echo "Job ID: $SLURM_JOB_ID"
echo "Job Name: $SLURM_JOB_NAME"
echo "Node: $SLURM_NODELIST"
echo "Start Time: $(date)"
echo "=========================================="
echo ""

# Navigate to project directory
cd /home/people/{student_number}/MRNet-assignment

echo "Current working directory: $(pwd)"
echo ""

# Activate virtual environment
source venv/bin/activate

# Create output directory for logs
mkdir -p eval_output
mkdir -p results

echo "=========================================="
echo "Starting Evaluation Job at $(date)"
echo "=========================================="
echo ""

# Run evaluation for ACL with both uncropped and cropped data
python3 code/src/evaluation.py \
    --condition acl \
    --architectures baseline \
    --data_modes uncropped cropped \
    --data_dir code/src/data/ \
    --checkpoint_dir checkpoints/ \
    --output_dir results/ \
    --n_bootstraps 1000

JOB_EXIT_CODE=$?

echo ""
echo "=========================================="
echo "Evaluation completed at $(date)"
echo "Exit code: $JOB_EXIT_CODE"
echo "=========================================="

#!/bin/bash -l
#SBATCH --job-name=eval_uncropped

#SBATCH --partition=gpu

#SBATCH --ntasks-per-node=4

#SBATCH --output=eval_output/eval_all_%j.log
#SBATCH --error=eval_output/eval_all_%j.err

echo "=========================================="
echo "Job ID: $SLURM_JOB_ID"
echo "Evaluating ALL UNCROPPED Models"
echo "Start Time: $(date)"
echo "=========================================="
echo ""

cd /home/people/{student_number}/MRNet-assignment
source venv/bin/activate
mkdir -p eval_output
mkdir -p results

echo "=========================================="
echo "Evaluating: ACL, Meniscus, Abnormal"
echo "Data modes: Uncropped, Cropped"
echo "Architecture: Baseline (AlexNet-ImageNet)"
echo "=========================================="
echo ""

# Run evaluation for ALL conditions
python3 code/src/evaluation.py \
    --condition all \
    --architectures baseline \
    --data_modes uncropped cropped \
    --data_dir code/src/data/ \
    --checkpoint_dir checkpoints/ \
    --output_dir results/ \
    --n_bootstraps 1000 \
    --seed 42

echo ""
echo "=========================================="
echo "All evaluations completed at $(date)"
echo "Exit code: $JOB_EXIT_CODE"
