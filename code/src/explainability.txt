#!/bin/bash -l
#SBATCH --job-name=explainability
# Request 1 node
#SBATCH -N 1
# GPU partition
#SBATCH --partition=gpu
# Request 1 GPU 
#SBATCH --gres=gpu:1
# Number of CPU cores
#SBATCH --ntasks-per-node=4
# Expected runtime: 1 hours 
#SBATCH -t 01:00:00
# Email notifications
#SBATCH --mail-type=ALL
#SBATCH --mail-user=student@ucdconnect.ie
# Output logs
#SBATCH --output=explainability_output/explain_%j.log
#SBATCH --error=explainability_output/explain_%j.err

# Print job information
echo "=========================================="
echo "Job ID: $SLURM_JOB_ID"
echo "Job Name: $SLURM_JOB_NAME"
echo "Node: $SLURM_NODELIST"
echo "Start Time: $(date)"
echo "=========================================="
echo ""

# Navigate to project directory
cd /home/people/{student_number}/MRNet-assignment

echo "Current working directory: $(pwd)"
echo ""

# Activate virtual environment
source venv/bin/activate

mkdir -p explainability_output

# Print Python and CUDA info
echo "Python version: $(python3 --version)"
echo "PyTorch version: $(python3 -c 'import torch; print(torch.__version__)')"
echo "CUDA available: $(python3 -c 'import torch; print(torch.cuda.is_available())')"
echo "CUDA version: $(python3 -c 'import torch; print(torch.version.cuda)')"
echo "Number of GPUs: $(python3 -c 'import torch; print(torch.cuda.device_count())')"
if python3 -c 'import torch; exit(0 if torch.cuda.is_available() else 1)'; then
    echo "GPU Name: $(python3 -c 'import torch; print(torch.cuda.get_device_name(0))')"
fi
echo ""

echo "=========================================="
echo "Starting Explainability Job at $(date)"
echo "=========================================="
echo ""

# Run script (include cropped or uncropped data)
python3 code/src/explainability.py \
  --checkpoint checkpoints/meniscus_sagittal_baseline_cropped_best.pth \
  --data_mode cropped \
  --data_dir src/data \
  --output eval_output/acl_sagittal_triage_report.html

JOB_EXIT_CODE=$?

echo ""
echo "=========================================="
echo "Explainability completed at $(date)"
echo "Exit code: $JOB_EXIT_CODE"
echo "=========================================="
echo ""

# Copy outputs to a timestamped folder for record-keeping
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
