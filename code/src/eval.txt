# Multi condition / Plane
# E.g. for uncropped 

#!/bin/bash -l
#SBATCH --job-name=eval_uncropped
#SBATCH -N 1
#SBATCH --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --ntasks-per-node=4
#SBATCH -t 02:00:00  # 2 hours for 9 evaluations
#SBATCH --mail-type=ALL
#SBATCH --mail-user=student@ucdconnect.ie
#SBATCH --output=eval_output/eval_uncropped_%j.log
#SBATCH --error=eval_output/eval_uncropped_%j.err

echo "=========================================="
echo "Job ID: $SLURM_JOB_ID"
echo "Evaluating ALL UNCROPPED Models"
echo "Start Time: $(date)"
echo "=========================================="
echo ""

cd /home/people/{student_number}/MRNet-assignment
source venv/bin/activate
mkdir -p eval_output

CONDITIONS=("acl" "meniscus" "abnormal")
PLANES=("sagittal" "axial" "coronal")
DATA_MODE="uncropped"

for condition in "${CONDITIONS[@]}"; do
  for plane in "${PLANES[@]}"; do
    
    CHECKPOINT="checkpoints/${condition}_${plane}_baseline_${DATA_MODE}_best.pth"
    
    echo ""
    echo "=========================================="
    echo "Evaluating: $condition - $plane - $DATA_MODE"
    echo "Checkpoint: $CHECKPOINT"
    echo "$(date)"
    echo "=========================================="
    
    python3 code/src/minimal_eval.py \
      --condition $condition \
      --plane $plane \
      --data_mode $DATA_MODE \
      --checkpoint $CHECKPOINT
    
    EXIT_CODE=$?
    if [ $EXIT_CODE -ne 0 ]; then
      echo "ERROR: Evaluation failed for $condition-$plane-$DATA_MODE"
    else
      echo "SUCCESS: Completed $condition-$plane-$DATA_MODE"
    fi
    
  done
done

echo ""
echo "=========================================="
echo "All UNCROPPED evaluation completed"
echo "End Time: $(date)"
echo "=========================================="
