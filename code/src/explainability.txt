#!/bin/bash -l
#SBATCH --job-name=multiplane_all
#SBATCH -N 1
#SBATCH --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --ntasks-per-node=4
#SBATCH -t 01:30:00  # 1.5 hours for all conditions
#SBATCH --mail-type=ALL
#SBATCH --mail-user=student@ucdconnect.ie
#SBATCH --output=explainability_output/multiplane_all_%j.log
#SBATCH --error=explainability_output/multiplane_all_%j.err

echo "=========================================="
echo "Job ID: $SLURM_JOB_ID"
echo "Multi-Plane Explainability - ALL CONDITIONS"
echo "Start Time: $(date)"
echo "=========================================="
echo ""

cd /home/people/{student_number}/MRNet-assignment
source venv/bin/activate

mkdir -p explainability_output
mkdir -p triage_reports

echo "Current working directory: $(pwd)"
echo ""

# Print system info
echo "Python: $(python3 --version)"
echo "GPU: $(python3 -c 'import torch; print(torch.cuda.get_device_name(0) if torch.cuda.is_available() else "CPU")')"
echo ""

CONDITIONS=("acl" "meniscus" "abnormal")
DATA_MODES=("uncropped" "cropped")

for condition in "${CONDITIONS[@]}"; do
  for data_mode in "${DATA_MODES[@]}"; do
    
    echo ""
    echo "=========================================="
    echo "Processing: ${condition^^} - $data_mode"
    echo "$(date)"
    echo "=========================================="
    
    OUTPUT_FILE="triage_reports/${condition}_${data_mode}_multiplane.html"
    
    python3 code/src/explainability_multiplane.py \
        --triage_json results/${condition}_triage.json \
        --results_json results/results_${condition}.json \
        --significance_json results/significance_${condition}.json \
        --condition $condition \
        --architecture baseline \
        --data_mode $data_mode \
        --checkpoint_dir checkpoints/ \
        --data_dir code/src/data/ \
        --output_html $OUTPUT_FILE \
        --top_n 20
    
    EXIT_CODE=$?
    if [ $EXIT_CODE -ne 0 ]; then
      echo "ERROR: Failed for $condition-$data_mode"
    else
      echo "SUCCESS: Generated $OUTPUT_FILE"
      ls -lh $OUTPUT_FILE
    fi
    
  done
done

echo ""
echo "=========================================="
echo "All triage reports completed at $(date)"
echo "=========================================="
echo ""

echo "Generated reports:"
ls -lh triage_reports/
echo ""
echo "Total files: $(ls triage_reports/*.html | wc -l)"
