#### Loading Logic
```python
import json
import pandas as pd

# Load patient IDs from JSON
with open(split_json) as f:
    patient_ids = json.load(f)[split]

# Load and filter labels
df = pd.read_csv(label_csv, header=0)
df = df[df.iloc[:, 0].isin(patient_ids)]

# Build file paths
for patient_id, label in df.itertuples(index=False):
    filename = f"{int(patient_id):04d}.npy"
    path = os.path.join(data_dir, filename)
    self.file_paths.append(path)
    self.labels.append(label)

# Calculate pos_weight (training split only)
if split == 'train':
    num_positive = sum(self.labels)
    num_negative = len(self.labels) - num_positive
    self.pos_weight = torch.tensor(num_negative / num_positive)
    print(f"[{condition}-{plane}-{split}] pos_weight: {self.pos_weight:.3f} "
          f"({num_positive} pos, {num_negative} neg)")
else:
    self.pos_weight = torch.tensor(1.0)
```

#### `__getitem__` Implementation
```python
import numpy as np
import torch

def __getitem__(self, idx):
    volume = np.load(self.file_paths[idx])
    
    if volume.ndim == 4 and volume.shape[1] == 3:
        volume = volume[:, 0, :, :]
    
    volume = volume.astype(np.float32)
    volume = (volume - volume.min()) / (volume.max() - volume.min() + 1e-8)
    
    if self.transform:
        volume = np.stack([self.transform(slice) for slice in volume])
    
    exam_tensor = torch.from_numpy(volume).float()
    exam_tensor = exam_tensor.unsqueeze(1).repeat(1, 3, 1, 1)
    
    label = torch.tensor(self.labels[idx], dtype=torch.float32)
    return exam_tensor, label
```

#### Augmentation Function
```python
import random
import numpy as np
from scipy.ndimage import rotate

def get_augmentation_transform(is_training=True):
    """Medical imaging augmentations (brightness/contrast jitter, rotations ±10°)."""
    if is_training:
        def augment(slice_2d):
            if random.random() > 0.5:
                alpha = random.uniform(0.8, 1.2)
                beta = random.uniform(-0.1, 0.1)
                slice_2d = np.clip(alpha * slice_2d + beta, 0, 1)
            
            if random.random() > 0.5:
                angle = random.uniform(-10, 10)
                slice_2d = rotate(slice_2d, angle, reshape=False, mode='nearest')
            
            return slice_2d
        return augment
    else:
        return lambda x: x
```

**Key Decisions:**
- ✅ Augment single channel, then stack to 3 channels
- ✅ [0, 1] normalization (min-max)
- ✅ No PIL conversion (numpy/tensor operations)
- ✅ Brightness/contrast jitter + small rotations (±10°)
- ❌ NO flips (preserve anatomical sidedness)
- ❌ NO random crops (conflicts with R-CNN preprocessing)

**Attributes:**
- `dataset.pos_weight` (torch.Tensor): For BCEWithLogitsLoss

### 2.2 Role 2: Baseline Models

**File:** `src/modules/baseline_models.py`

**Class:** `MRNetBaseModel(nn.Module)`

**Architecture:**
```
Input: (batch_size, num_slices, 3, H, W)
  ↓
ResNet18 backbone (no final FC): (num_slices, 3, H, W) → (num_slices, 512)
  ↓
Max pooling across slices: (num_slices, 512) → (512,)
  ↓
FC head: (512,) → (1,) logit
```

**Implementation:**

#### Constructor
```python
import torch
import torch.nn as nn

class MRNetBaseModel(nn.Module):
    def __init__(self, backbone):
        super().__init__()
        self.backbone = backbone
        self.fc = nn.Linear(512, 1)
    
    def forward(self, x):
        """
        Args: x (batch_size, num_slices, 3, H, W)
        Returns: logits (batch_size, 1)
        """
        batch_size, num_slices = x.shape[0], x.shape[1]
        
        x = x.view(batch_size * num_slices, 3, x.shape[3], x.shape[4])
        features = self.backbone(x)
        features = features.view(features.size(0), -1)
        features = features.view(batch_size, num_slices, -1)
        
        pooled, _ = torch.max(features, dim=1)
        logits = self.fc(pooled)
        return logits
    
    def forward_with_slice_tracking(self, x):
        """
        Extended forward pass for explainability.
        Returns: logits (batch_size, 1), slice_indices (batch_size, 512)
        """
        batch_size, num_slices = x.shape[0], x.shape[1]
        
        x = x.view(batch_size * num_slices, 3, x.shape[3], x.shape[4])
        features = self.backbone(x)
        features = features.view(features.size(0), -1)
        features = features.view(batch_size, num_slices, -1)
        
        pooled, slice_indices = torch.max(features, dim=1)
        logits = self.fc(pooled)
        return logits, slice_indices
```

#### Factory Function
```python
import torch.nn as nn
import torchvision.models as models

def create_baseline_model():
    """Create baseline ResNet18 model with ImageNet pretraining."""
    resnet18 = models.resnet18(pretrained=True)
    backbone = nn.Sequential(*list(resnet18.children())[:-1])
    model = MRNetBaseModel(backbone)
    return model
```

**Key Decisions:**
- ✅ ResNet18 → 512 features → FC(512→1)
- ✅ Single-phase training (all parameters unfrozen from start)
- ✅ Slice tracking: Returns full (512,) indices tensor for explainability
- ✅ Max pooling: `torch.max(features, dim=1)` across slices

**Note:** Freeze/unfreeze methods exist in base class but NOT used for single-phase training.

parser.add_argument('--condition', type=str, required=True, choices=['acl', 'meniscus', 'abnormal'])
parser.add_argument('--plane', type=str, required=True, choices=['axial', 'coronal', 'sagittal'])

**Training Loop (Minimal):**
```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader
from sklearn.metrics import roc_auc_score

def main():
    args = parse_args()
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    
    train_dataset = MRNetDataset(
        root_dir=args.data_dir,
        condition=args.condition,
        plane=args.plane,
        split='train',
        data_mode=args.data_mode,
        transform=get_augmentation_transform(is_training=True)
    )
    val_dataset = MRNetDataset(
        root_dir=args.data_dir,
        condition=args.condition,
        plane=args.plane,
        split='val',
        data_mode=args.data_mode,
        transform=None
    )
    
    train_loader = DataLoader(train_dataset, batch_size=1, shuffle=True)
    val_loader = DataLoader(val_dataset, batch_size=1, shuffle=False)
    
    model = create_baseline_model().to(device)
    criterion = nn.BCEWithLogitsLoss(pos_weight=train_dataset.pos_weight.to(device))
    optimizer = optim.Adam(model.parameters(), lr=args.lr)
    
    best_auc = 0.0
    for epoch in range(args.epochs):
        model.train()
        train_loss = 0.0
        for exam, label in train_loader:
            exam, label = exam.to(device), label.to(device)
            
            optimizer.zero_grad()
            logits = model(exam)
            loss = criterion(logits, label.unsqueeze(1))
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
        
        model.eval()
        val_preds, val_labels = [], []
        with torch.no_grad():
            for exam, label in val_loader:
                exam = exam.to(device)
                logits = model(exam)
                probs = torch.sigmoid(logits).cpu().numpy()
                val_preds.extend(probs)
                val_labels.extend(label.numpy())
        
        val_auc = roc_auc_score(val_labels, val_preds)
        
        print(f"Epoch {epoch+1}/{args.epochs} - "
              f"Train Loss: {train_loss/len(train_loader):.4f}, "
              f"Val AUC: {val_auc:.4f}")
        
        if val_auc > best_auc:
            best_auc = val_auc
            checkpoint_path = f"{args.checkpoint_dir}/{args.condition}_{args.plane}_{args.architecture}_{args.data_mode}_best.pth"
            torch.save({
                'epoch': epoch,
                'model_state_dict': model.state_dict(),
                'optimizer_state_dict': optimizer.state_dict(),
                'val_auc': val_auc,
                'args': vars(args)
            }, checkpoint_path)
            print(f"  ✓ Saved best checkpoint (AUC: {val_auc:.4f})")
    
    print(f"\nTraining complete! Best Val AUC: {best_auc:.4f}")
```

**Key Decisions:**
- ✅ Single-phase training (all params unfrozen from start, lr=1e-4)
- ✅ Minimal logging (print statements only, no TensorBoard)
- ✅ Best checkpoint only (based on validation AUC)
- ✅ Configurable hyperparameters via command-line args
- ❌ NO early stopping, LR scheduling (defer to Role 4)

### 3.2 Minimal Explainability

**File:** `src/explainability.py` (already exists, needs implementation)

**Scope:** ACL condition, sagittal plane, top-20 cases (10 positive, 10 negative)

**HTML Visualization Components (for each case):**
1. Original most informative slice image (grayscale)
2. Grad-CAM heatmap overlaid on the same slice
3. Overall model confidence (final prediction probability)
4. Slice contribution count (how many of 512 features came from this slice)

**Implementation:**

```python
import torch
import numpy as np
import matplotlib.pyplot as plt
import base64
from io import BytesIO
from scipy.stats import mode
from pytorch_grad_cam import GradCAM
from pytorch_grad_cam.utils.image import show_cam_on_image

def get_gradcam_heatmap(model, exam, target_slice_idx):
    """Generate Grad-CAM heatmap for a specific slice."""
    target_layers = [model.backbone[-2]]
    cam = GradCAM(model=model, target_layers=target_layers)
    
    # Extract single slice and generate heatmap
    slice_input = exam[:, target_slice_idx:target_slice_idx+1, :, :, :]
    grayscale_cam = cam(input_tensor=slice_input)
    
    return grayscale_cam[0]

def generate_triage_html(model, test_dataset, device, output_path='triage_report.html'):
    """Generate static HTML with top-20 cases showing 4 visualization components."""
    model.eval()
    predictions = []
    
    for idx in range(len(test_dataset)):
        exam, label = test_dataset[idx]
        exam_input = exam.unsqueeze(0).to(device)
        
        with torch.no_grad():
            logits, slice_indices = model.forward_with_slice_tracking(exam_input)
            prob = torch.sigmoid(logits).item()
        
        most_informative_slice = mode(slice_indices.cpu().numpy(), axis=1)[0][0]
        contribution_count = (slice_indices == most_informative_slice).sum().item()
        
        predictions.append({
            'idx': idx,
            'true_label': label.item(),
            'pred_prob': prob,
            'slice_idx': most_informative_slice,
            'contribution': contribution_count
        })
    
    predictions.sort(key=lambda x: x['pred_prob'], reverse=True)
    top_positive = [p for p in predictions if p['pred_prob'] > 0.5][:10]
    top_negative = [p for p in predictions if p['pred_prob'] <= 0.5][-10:]
    top_cases = top_positive + top_negative
    
    html = ['<html><head><title>ACL Triage Report</title>']
    html.append('<style>body{font-family:Arial;} .case{margin:30px;border:1px solid #ccc;padding:20px;}</style>')
    html.append('</head><body>')
    html.append('<h1>ACL Detection - Sagittal Plane - Top 20 Cases</h1>')
    
    for case in top_cases:
        exam, label = test_dataset[case['idx']]
        slice_idx = case['slice_idx']
        
        original_slice = exam[slice_idx, 0, :, :].numpy()
        heatmap = get_gradcam_heatmap(model, exam.unsqueeze(0).to(device), slice_idx)
        
        overlay = show_cam_on_image(
            np.stack([original_slice]*3, axis=2), 
            heatmap, 
            use_rgb=True
        )
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 5))
        ax1.imshow(original_slice, cmap='gray')
        ax1.set_title('Original Slice')
        ax1.axis('off')
        ax2.imshow(overlay)
        ax2.set_title('Grad-CAM Overlay')
        ax2.axis('off')
        
        buf = BytesIO()
        plt.savefig(buf, format='png', bbox_inches='tight')
        plt.close()
        img_b64 = base64.b64encode(buf.getvalue()).decode()
        
        contribution_pct = (case['contribution'] / 512) * 100
        
        html.append('<div class="case">')
        html.append(f'<h3>Case #{case["idx"]}</h3>')
        html.append(f'<p><strong>Overall Confidence:</strong> {case["pred_prob"]:.3f} ({case["pred_prob"]*100:.1f}%)</p>')
        html.append(f'<p><strong>True Label:</strong> {"Positive (ACL Tear)" if case["true_label"] == 1 else "Negative (No Tear)"}</p>')
        html.append(f'<p><strong>Most Informative Slice:</strong> #{slice_idx}</p>')
        html.append(f'<p><strong>Slice Contribution:</strong> {case["contribution"]}/512 features ({contribution_pct:.1f}%)</p>')
        html.append(f'<img src="data:image/png;base64,{img_b64}" style="width:100%;max-width:800px;"/>')
        html.append('</div>')
    
    html.append('</body></html>')
    
    with open(output_path, 'w') as f:
        f.write('\n'.join(html))
    
    print(f"Triage report saved to {output_path}")
```
